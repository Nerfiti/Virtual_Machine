FillVideo:
	pop RDX

	push RDX
	push 1
	sub
	pop RDX
	pop [300 + RDX]
	push RDX
	push 0
	jbe endFunc:
	jmp repeat:
	endFunc:
	ret

	push 0

	jmp two_ans:
